JavaScript Code Example

<script>
  if(document.getElementById("ttrs-hack-panel")) return;
  let intervalId = null, delay = 600, answerCount = 0, maxAnswers = Infinity, isPaused = false;

  function createPanel() {
    let panel = document.createElement("div");
    panel.id = "ttrs-hack-panel";
    panel.style = "position:fixed;bottom:20px;right:20px;z-index:9999;background:#f4f4f4;color:#333;padding:10px 12px;border:1px solid #ccc;border-radius:10px;font-family:sans-serif;font-size:13px;box-shadow:0 4px 8px rgba(0,0,0,0.1);min-width:180px;cursor:move;user-select:none;";
    panel.innerHTML = `
      <div id="ttrs-header" style="font-weight:600;margin-bottom:5px;cursor:move;">⚙️ TTRS Helper</div>
      <div style="margin-bottom:5px;">Answers: <span id="ttrs-count">0</span></div>
      <div style="margin-bottom:5px;">
        <button id="ttrs-start" style="margin-right:4px;">Start</button>
        <button id="ttrs-stop">Stop</button>
        <button id="ttrs-pause" style="margin-left:4px;">Pause</button>
      </div>
      <div style="margin-bottom:5px;">
        <label>How much help (ms):</label>
        <input id="ttrs-help" type="number" value="600" style="width:50px;">
      </div>
      <div style="margin-bottom:5px;">
        <label>Max Answers:</label>
        <input id="ttrs-max" type="number" value="0" style="width:50px;">
      </div>
      <div style="margin-bottom:5px;">
        <label>Theme:</label>
        <select id="ttrs-theme">
          <option value="light">Light</option>
          <option value="dark">Dark</option>
        </select>
      </div>
      <button id="ttrs-hide" style="font-size:11px;">Hide Panel</button>
    `;
    document.body.appendChild(panel);
  }

  function startHack() {
    if(intervalId) clearInterval(intervalId);
    answerCount = 0;
    intervalId = setInterval(getAnswer, delay);
  }

  function stopHack() {
    clearInterval(intervalId);
    intervalId = null;
  }

  function getAnswer() {
    if (isPaused) return;
    if (answerCount >= maxAnswers) {
      stopHack();
      alert("Reached max answers: " + maxAnswers);
      return;
    }
    let q = document.querySelector("ttr-game-question:nth-child(2)");
    if (!q) return;
    let t = q.innerText.trim().replace("×", "*").replace("÷", "/");
    let a;
    try {
      let parts = t.match(/(\d+)\s*([\*\/])\s*(\d+)/);
      if (!parts) return;
      a = parts[2] === "*" ? (parseInt(parts[1]) * parseInt(parts[3])) : (parseInt(parts[1]) / parseInt(parts[3]));
      a = Math.floor(a);
    } catch (err) {
      console.error("Parse error", err);
      return;
    }
    let keys = [
      document.querySelector("ttr-game-numpad > div > div:nth-child(4) > div.key-n.key-shade"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(1) > div:nth-child(1)"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(1) > div:nth-child(2)"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(1) > div:nth-child(3)"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(2) > div:nth-child(1)"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(2) > div:nth-child(2)"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(2) > div:nth-child(3)"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(3) > div:nth-child(1)"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(3) > div:nth-child(2)"),
      document.querySelector("ttr-game-numpad > div > div:nth-child(3) > div:nth-child(3)")
    ];
    let enter = document.querySelector("ttr-game-numpad > div > div:nth-child(4) > div.key-ent.ng-star-inserted");
    if (!enter) return;
    a.toString().split('"').forEach(function(c) {
      let k = keys[parseInt(c)];
      if (k) k.click();
    });
    enter.click();
    answerCount++;
  }

  createPanel();
</script>
